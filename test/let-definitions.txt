# Simple Let Definition with double terminator

let = 123.. ..

==>

GraffiticodeText(Program(LetDefinition(Let,"=",Expression(Number),Terminator),Terminator))

# Let with String Value with double terminator

let = "hello world".. ..

==>

GraffiticodeText(Program(LetDefinition(Let,"=",Expression(String),Terminator),Terminator))

# Let with Object with double terminator

let = {
  "name": "value",
  "nested": {
    "x": 1,
    "y": 2,
  },
}.. ..

==>

GraffiticodeText(Program(LetDefinition(Let,"=",Expression(Object(
  Property(PropertyName,Expression(String)),
  Property(PropertyName,Expression(Object(
    Property(PropertyName,Expression(Number)),
    Property(PropertyName,Expression(Number))))))),Terminator),Terminator))

# Let with Array with double terminator

let = [1, "two", true, null].. ..

==>

GraffiticodeText(Program(LetDefinition(Let,"=",Expression(Array(
  Expression(Number),
  Expression(String),
  Expression(True),
  Expression(Null))),Terminator),Terminator))

# Multiple Statements with terminator for each statement and program

let = 42..
{
  "foo": "bar"
}
[1, 2, 3]..

==>

GraffiticodeText(Program(
  LetDefinition(Let,"=",Expression(Number),Terminator),
  ExpressionStatement(Expression(Object(Property(PropertyName,Expression(String))))),
  ExpressionStatement(Expression(Array(
    Expression(Number),
    Expression(Number),
    Expression(Number)))),Terminator))
