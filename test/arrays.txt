# Empty Array

[ ]..

==>

GraffiticodeText(Program(ExpressionStatement(Expression(Array)),Terminator))

# Array With One Value

["One is the loneliest number"]..

==>

GraffiticodeText(Program(ExpressionStatement(Expression(Array(Expression(String)))),Terminator))

# Array With Multiple Values

[
  "The more the merrier",
  1e5,
  true,
  { },
  ["I'm", "nested"]
]..

==>

GraffiticodeText(Program(ExpressionStatement(Expression(Array(
  Expression(String),
  Expression(Number),
  Expression(True),
  Expression(Object),
  Expression(Array(Expression(String),Expression(String)))))),Terminator))

==>

GraffiticodeText(Program(ExpressionStatement(Expression(Array(

  Expression(Array(Expression(String),Expression(String)))))),Terminator))
